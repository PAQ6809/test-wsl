  async function fileFingerprint(file){const sample=1024*1024;const first=file.slice(0,Math.min(sample,file.size));const last=file.slice(Math.max(0,file.size-sample),file.size);return sha256Hex(new Blob([first,last,new TextEncoder().encode(`${file.name}|${file.size}|${file.lastModified}`)]))}
