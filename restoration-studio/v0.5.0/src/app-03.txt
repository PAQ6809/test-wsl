  async function waitForBuffer(jobId,totalSize,signal){

  }
  async function runCloudRelay(file,signal){

    for(let part=1;part<=job.total_parts;part++){
      if(signal.aborted)throw new DOMException("Aborted","AbortError");if(done.has(part))continue;await waitForBuffer(job.id,file.size,signal);
      const start=(part-1)*C.relayChunkBytes,end=Math.min(file.size,start+C.relayChunkBytes),blob=file.slice(start,end);const hash=await sha256Hex(blob);const objectPath=`${state.user.id}/${job.id}/input/${String(part).padStart(6,"0")}.part`;
      const upload=await sb.storage.from(C.storageBucket).upload(objectPath,blob,{contentType:"application/octet-stream",upsert:false,cacheControl:"0"});if(upload.error&&!String(upload.error.message).includes("already exists"))throw upload.error;
      const row=await sb.from("restoration_upload_parts").upsert({job_id:job.id,user_id:state.user.id,part_number:part,object_path:objectPath,size:blob.size,sha256:hash,status:"uploaded"},{onConflict:"job_id,part_number"});if(row.error)throw row.error;
      showUpload(`安全上傳第 ${part}/${job.total_parts} 段`,end/file.size*100);
    }

  }

  async function runBrowserImage(file,signal){

    showUpload("解碼與高品質放大",12);const target=Math.max(bitmap.height,Number($("targetHeight").value));const scale=target/bitmap.height;const width=Math.round(bitmap.width*scale);const canvas=document.createElement("canvas");canvas.width=width;canvas.height=target;const ctx=canvas.getContext("2d",{willReadFrequently:true});ctx.imageSmoothingEnabled=true;ctx.imageSmoothingQuality="high";ctx.drawImage(bitmap,0,0,width,target);bitmap.close();

    let quality=.95;let blob=await new Promise(r=>canvas.toBlob(r,"image/jpeg",quality));while(blob.size>30*1024**2&&quality>.72){quality-=.05;blob=await new Promise(r=>canvas.toBlob(r,"image/jpeg",quality))}if(!blob)throw new Error("無法建立圖片成品");

  async function processPixels(imageData,strength){
    if(!window.Worker)return imageData;
    const width=imageData.width,height=imageData.height,pixels=imageData.data.buffer;
    return new Promise((resolve,reject)=>{const worker=new Worker("/image-worker.js");worker.onmessage=e=>{worker.terminate();resolve(new ImageData(new Uint8ClampedArray(e.data.pixels),width,height))};worker.onerror=e=>{worker.terminate();reject(e)};worker.postMessage({pixels,width,height,strength},[pixels])})
  }
